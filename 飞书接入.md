架构
飞书 App
  │ 用户发消息
  ▼
飞书 Bot（Webhook）
  │ HTTP POST
  ▼
OpenClaw 服务器 ── WebSocket/MQTT ──→ ESP32（TTS 语音 + 屏幕显示）
  │
  ▼
HTTP Reply → 飞书 Bot → 飞书 App（文字回复）

# feishu_handler.py — 飞书消息处理模块

import json
import hashlib
import logging
from typing import Optional

logger = logging.getLogger(__name__)

class FeishuBot:
    """飞书 Bot 消息处理 — 桥接到小智 AI 的 LLM + TTS 流水线"""

    def__init__(self, app_id: str, app_secret: str, verify_token: str):
        self.app_id = app_id
        self.app_secret = app_secret
        self.verify_token = verify_token
        self._tenant_access_token: Optional[str] = None

    # ========== 飞书事件接收 ==========

    def verify_url(self, challenge: str, token: str) -> dict:
        """URL 验证（飞书首次配置回调地址时调用）"""
        if token != self.verify_token:
            return {"code": 1, "msg": "invalid token"}
        return {"challenge": challenge}

    async def handle_event(self, event: dict) -> Optional[dict]:
        """处理飞书推送事件"""
        event_type = event.get("type", "")

    if event_type == "url_verification":
            return self.verify_url(
                event.get("challenge", ""),
                event.get("token", "")
            )

    if event_type == "im.message.receive_v1":
            # 只处理文本消息
            msg_type = event.get("event", {}).get("message", {}).get("message_type", "")
            if msg_type != "text":
                return None

    content = json.loads(
                event["event"]["message"]["content"]
            )
            user_text = content.get("text", "")

    # 过滤掉 @bot 的前缀
            if user_text.startswith("@_bot"):
                user_text = user_text[len("@_bot"):].strip()
            if user_text.startswith("@_all"):
                user_text = user_text[len("@_all"):].strip()

    if not user_text:
                return None

    logger.info(f"[飞书] 收到消息: {user_text}")

    # 异步处理（先返回 200，避免飞书超时重试）
            import asyncio
            asyncio.create_task(self._process_message(
                user_text=user_text,
                chat_id=event["event"]["message"]["chat_id"],
                message_id=event["event"]["message"]["message_id"]
            ))
            return None  # 不需要同步返回

    # ========== 核心：文字 → LLM → 双路回复 ==========

    async def _process_message(self, user_text: str, chat_id: str, message_id: str):
        """处理用户文字：调用 LLM + TTS，双路回复飞书和 ESP32"""

    # 1. 调用 LLM（直接传文字，跳过 ASR）
        llm_response = await self._call_llm(user_text)

    # 2. 回复飞书（文字）
        await self._reply_feishu_text(chat_id, message_id, llm_response.text)

    # 3. 推送给 ESP32（TTS 语音 + 屏幕显示）
        await self._push_to_esp32(llm_response.text)

    async def _call_llm(self, text: str) -> "LLMResponse":
        """调用 OpenClaw 现有的 LLM 流水线"""
        # 复用 OpenClaw 已有的 LLM 调用逻辑
        # 此处伪代码，实际调用你们的 LLM service
        from services.llm_service import LLMService
        llm = LLMService()
        return await llm.chat(user_text=text)

    async def _reply_feishu_text(self, chat_id: str, root_message_id: str, text: str):
        """飞书回复文字消息（作为话题回复）"""
        url = "https://open.feishu.cn/open-apis/im/v1/messages/reply"
        token = await self._get_tenant_access_token()

    payload = {
            "msg_type": "text",
            "content": json.dumps({"text": text}),
        }

    import aiohttp
        async with aiohttp.ClientSession() as session:
            async with session.post(
                url,
                params={"receive_id_type": "chat_id"},
                headers={
                    "Authorization": f"Bearer {token}",
                    "Content-Type": "application/json",
                },
                json=payload,
            ) as resp:
                result = await resp.json()
                if result.get("code") != 0:
                    logger.error(f"[飞书] 发送回复失败: {result}")

    async def _push_to_esp32(self, tts_text: str):
        """推送 TTS 到 ESP32 设备"""
        # 复用 OpenClaw 现有的 TTS → ESP32 推送逻辑
        # 这与语音场景下 LLM 回复后推 TTS 完全一样
        from services.tts_service import TTSService
        tts = TTSService()

    async for sentence, audio_chunk in tts.generate(tts_text):
            # 发送 JSON 给 ESP32（屏幕显示文字）
            await self._send_to_esp32_json({
                "type": "tts",
                "state": "sentence_start",
                "text": sentence
            })
            # 发送音频给 ESP32（喇叭播放）
            await self._send_to_esp32_binary(audio_chunk)

    # TTS 结束信号
        await self._send_to_esp32_json({
            "type": "tts",
            "state": "stop"
        })

    # ========== 飞书 Token 管理 ==========

    async def _get_tenant_access_token(self) -> str:
        """获取 tenant_access_token（带缓存，有效期 2 小时）"""
        if self._tenant_access_token:
            return self._tenant_access_token

    import aiohttp

            async with session.post(url, json={
                "app_id": self.app_id,

            }) as resp:
                result = await resp.json()
                self._tenant_access_token = result["tenant_access_token"]
                return self._tenant_access_token

# 在 main.py 或 routers/ 中添加

from fastapi import FastAPI, Request
from feishu_handler import FeishuBot

app = FastAPI()

# 从配置读取

feishu = FeishuBot(
    app_id=config.FEISHU_APP_ID,
    app_secret=config.FEISHU_APP_SECRET,
    verify_token=config.FEISHU_VERIFY_TOKEN
)

@app.post("/feishu/callback")
async def feishu_callback(request: Request):
    """飞书事件回调入口"""
    body = await request.json()

    # 处理事件
    if "challenge" in body:
        # URL 验证
        return feishu.verify_url(
            body.get("challenge", ""),
            body.get("token", "")
        )

    # 处理消息事件
    result = await feishu.handle_event(body)
    if result:
        return result
    return {"code": 0}
